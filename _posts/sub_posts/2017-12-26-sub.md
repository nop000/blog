//冒泡排序
int *BubbleSort(int *ary, int length)
{
    int i, j, tmp;
    for (i=0; i<length-1; i++)
    {
        tmp = ary[i];
        for (j = length-1; j > i; j--)
        {
            //找到数组中最小的数，并交换
            if (tmp > ary[j])
            {
                ary[i] = ary[j];
                ary[j] = tmp;
                tmp = ary[i];
            }
        }
    }
    return ary;
}
